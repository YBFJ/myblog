                  # 1. 字符串中的字母变成驼峰式

                  JS 函数

                  function getCamelCase(str) {
  var arr = str.split("-");
  return arr
    .map(function(item, index) {
      console.log(item);
      if (index === 0) {
        return item;
      } else {
        return item.charAt(0).toUpperCase() + item.slice(1);
      }
    })
    .join("");
}

console.log(getCamelCase("xiaoshuo-ss-sfff-fe"));

                  正则写法：

                  function getCamelCase(str) {
  return str.replace(/-([a-z])/g, function(all, i) {
    // 括号中的会进行分组，每组为匹配到的东西
    // -s s
    //   - s s
    //     - f f
    console.log(all);
    console.log(i);
    return i.toUpperCase();
  });
}

console.log(getCamelCase("xiaoshuo-ss-sfff-fe"));

                  如果需要反向转化，将驼峰转化成有-的

                  function getCamelCase(str) {
  var arr = str.split("");
  return arr
    .map(function(item, index) {
      console.log(item);
      if (item.toUpperCase() === item) {
        return "-" + item.toLowerCase();
      } else {
        return item;
      }
    })
    .join("");
}

console.log(getCamelCase("xiaoshuoSsSfffFe"));

                  正则写法：

                  function getCamelCase(str) {

                  原生 JS 去重

                  // 传统去重写法
function unique(arr) {
  var n = [];
  for (var i = 0; i < arr.length; i++) {
    if (n.indexOf(arr[i]) === -1) {
      n.push(arr[i]);
    }
  }
  return n;
}
var arry = [1, 25, 15, 1, 2, 15, 5, 15, 25, 35, 1];
console.log(unique(arry));

                  # 3. 统计字符串中出现的最多的字母

                  其实就是先生成一个对象，将字母储存成 key，字母出现的次数储存成值； 再对对象进行遍历，找出最大的值及其对应的 key，也就是出现最多的字母了。

                  function getChar(str) {
  if (typeof str !== "string") return;
  const obj = new Object();
  for (let i = 0; i < str.length; i++) {
    let char = str.charAt(i);
    obj[char] = obj[char] || 0; // 不存在就是0
    obj[char]++;
  }
  let maxChar; // 存储出现最多的字母
  let maxNum = 0; // 存储maxChar字母出现的次数
  for (let key in obj) {
    if (obj[key] > maxNum) {
      maxChar = key; // 比较后存储次数多的字母
      maxNum = obj[key];
    }
  }
  return "出现次数最多的字母为：" + maxChar + ";" + "出现的次数为" + maxNum;
}

let str1 = "aabbbccdd";
getChar(str1);

                  # 5. 深拷贝

                  浅拷贝只需要 Object.assign()； 符合 json 要求（双引号）的对象深拷贝：

                  JSON.parse(JSON.stringfy(obj));

                  不严格符合 json 的对象的深拷贝：

                  function deepCopy(src, r) {
  var r = r || {};
  for (var prop in src) {
    console.log(prop);
    if (typeof src[prop] === "object") {
      if (src[prop].constructor === Array) {
        console.log("数组");
        r[prop] = [];
      } else {
        console.log("对象");
        r[prop] = {};
      }
      deepCopy(src[prop], r[prop]);
    } else {
      // 非对象直接拷贝
      // 此处就是浅拷贝
      r[prop] = src[prop];
    }
  }
  return r;
}

                  # 7. 合并多个数组排序

                  以下写法为升序，如果需要降序则在返回的结果前面加上-

                  var arr = [1, 25, 3, 85, 4, 0];
arr.sort((m, n) => {
  if (m > n) {
    return 1; // 顺序改变
  } else {
    return -1; // 顺序不改变
  }
});

console.log(arr);

                  同时也是可以简化成

                  var arr = [1, 25, 3, 85, 4, 0];
arr.sort((m, n) => {
  if (m > n) {
    return m - n;
  }
});
console.log(arr);

                  # 8. 约瑟夫环问题

                  编号为 1-100 的 100 人围成一个圈子，以 123123 的方式进行报数，疏导 3 的人自动退出圈子，剩下的人继续报数，问最后一个剩下的人编号是几？ 递归解法： 假设，n 为总人数，报数出环数字为 k（123123 的 3） i 为第 i 次出环 当 i = 1 时，f(n, k, i) = (n + k -1) %n 当 i! = 1 时，f(n, k, i) = (f(n-1,k,i-1) + k ) %n 简单来说，这一轮=（上一轮+k）%n 首先：是比较复杂的递归解法，编号为 1-100，报数逢 3 退出，注意解法中数组的下标是从 0 开始的， 经过我的手动验证确实是对的，但是原理我现在还不是很懂，先放着后续再弄。

                  // 总人数n，报数出环数字k n=100 k=3 第i个人出环
function josephus(n, k, i) {
  if (i == 1) {
    return (n + k - 1) % n;
  } else {
    return (josephus(n - 1, k, i - 1) + k) % n;
  }
}

for (var i = 1; i <= 100; i++) {
  console.log("第" + i + "次出环，下标为：" + josephus(100, 3, i));
}
// 第100次：下标为90的人出环 他的编号是91（需要加上1）

                  第二种解法： 比较通俗易懂，将编号组装成为一个数组，设置一个浮标 k，123 的循环。每次到 3 的时候将指向的值移除即可。 编号为 1-100，报数逢 3 退出。 这个逻辑也便于理解，并且最后返回来是正常的编号，不需要加一，因为编号直接在数组中，使用的时候使用的是编号的索引。

                  function josephus2(n, k) {
  var players = [];
  for (var i = 1; i <= n; i++) {
    players.push(i);
  }
  var flag = 0;
  while (players.length > 1) {
    var outNum = 0;
    // outNum为本次循环已出局人数，便于定位元素位置
    var len = players.length;
    for (var i = 0; i < len; i++) {
      flag++;
      if (flag == k) {
        flag = 0;
        console.log(
          "出局:" + players[i - outNum] + "本次循环已出局人数：" + outNum
        );
        // i循环时受数组长度影响 每从头开始时i变为0； 但flag始终是123..k  123..k 固定循环，不受数组影响
        players.splice(i - outNum, 1);
        outNum++;
      }
    }
  }
  return players[0];
}
console.log(josephus2(100, 3));
