生成器是一个带星号的函数，它并不是真正的函数，它可以通过 yield 关键字暂停执行和恢复执行

function* gen() {
  console.log("enter")

  return 3
}

# 生成器的实现机制----协程

协程是一直比线程更加轻量级的存在，协程处在线程的环境中，一个线程可以存在多个协程，不像进程和线程，协程并不受操作系统的管理，而是被具体的应用程序代码所控制。

一个线程一次只能执行一个写成。比如当前执行 A 协程，另外还有一个 B 协程，如果要执行 B 的任务，就必须要在 A 协程中将 JS 的线程控制权转交给 B 协程，B 执行，A 就相当于暂停的状态。 在这个过程中 A 将执行权交给了 B，也就是 A 启动了 B，也就说 A 是 B 的父协程；类似于上述代码中 gen1 可以说是 gen2 的父协程

# Async / await 逻辑

async/await 被称为 JS 中异步终极解决方案

# async

async 是一个通过异步执行，并隐式返回 Promise 作为结果的函数

# await

await 只能在 async 中被使用

  console.log(100)

  console.log(x)
  console.log(200)
}

let promise = new Promise((resolve, reject) => {

然后 JS 引擎 将暂停当前协程的运行，将线程的执行权交给了父协程 ​

回到父协程，父协程的第一件事情，就是对 await 返回的 Promise 调用 then，来监听这个 promise 的改变；当主线程的宏任务完成就去检查为任务队列，发现还有一个 Promsie 的 resolve

promise.then((value) => {
  // 1. 将线程的执行权交给test协程
  // 2. 把value值传给test协程
})

然后执行权交给了 test 协程手上，test 接收到父协程传来的 200，赋值给 a，然后依次执行后面的语句 200，200 ​

# 代码级别的理解

对于 async、await，其实 aync 就在是返回一个隐式的 promise，await 中需要检查是否在 async 里面，并且需要通过 while(true)把当前的这一套流程挂起，直到.next 完成；而真实的 async 和 await 其实就是在实现这个 while true

// 实际的使用中await只能在async中使用

function getGen() {
  let g = {
    done: false,
    count: 0,
    next() {
      if (this.count === 3) this.done = true
      if (this.done) return { value: this.count, done: this.done }
      this.count++
      return { value: this.count, done: this.done }
    },
  }
  return g
}

function myAwait() {
  this.c = function (data) {
    if (!this.isCalledByAsync) throw new Error("Should be called by async")
    let result = data.next()
    // 关键的挂起行为
    while (true) {
      console.log("awaiting...", result)
      // result.then((res) => console.log(res))
      console.log("resolveData", result.value)

      if (result.done) return result.value
      result = data.next()
    }
  }
}

function myAsync(fn) {
  myAwait.prototype.isCalledByAsync = true
  let m = fn()
  console.log("async before")
  let d = new myAwait().c(m)
  console.log("d", d)
  console.log("async after")
  myAwait.prototype.isCalledByAsync = false
}

function myAwait() {
  this.c = function (data) {
    if (!this.isCalledByAsync) throw Error("Should be called by async")
    // let result = data.next();
    // while(true){
    //     console.log('awaiting....', result);
    //     if(result.done) return result.value;
    //     result = data.next()
    // }
    return new Promise((resolve, reject) => {
      // 内部定义了一个step函数 用来一步步的跨过yield的阻碍
      // key有next和throw两种取值 分别对应gen的next和throw方法
      // arg参数则是用来把promise resolve出来的值交给下一个yield
      function step(key, arg) {
        let generatorResult
        // 这个方法需要包裹在try catch中
        // 如果出错就把primise给reject掉 外部通过.catch可以获取到错误
        try {
          generatorResult = data[key](arg)
        } catch (error) {
          return reject(error)
        }
        console.log("awaiting....", generatorResult)

        // gen.next()得到的结果是一个{value, done}的结构
        const { value, done } = generatorResult

        if (done) {
          return resolve(value)
        }
        return Promise.resolve(value).then(
          (val) => {
            step("next", val)
          },
          (err) => {
            step("throw", err)
          }
        )
      }
      step("next")
    })
  }
}

console.log("no async after")
